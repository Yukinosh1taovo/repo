[713. 乘积小于 K 的子数组](https://leetcode.cn/problems/subarray-product-less-than-k/)

```cpp
class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        // 时间复杂度O(n)
        // 空间复杂度O(1)
        if (k <= 1) //整数乘积不可能小于1
        {
            return 0;
        }
        int n = nums.size();
        int l = 0, r = l;
        int prod = 1;
        int ans = 0;
        while (r < n) // 枚举右端点
        {
            prod *= nums[r];

            while (prod >= k) //保证while结束后为满满足条件
            {
                prod /= nums[l];
                l++;
            }
            ans += r - l + 1; // [l, r], [l + 1, r] ... [r, r] 均为答案
            r++;
        }
        return ans;
    }
};
```
